%%%%%%%%%%%%%%%%%%%%%% Listings %%%%%%%%%%%%%%%%%%%%%%
\usepackage{listings} % til kildekode/nummererede linjer mv.
\lstset{
	language=java, % Sprog for listings
	morekeywords={String}, % Evt. ekstra Keywords
	inputencoding=latin1, % Encoding for inkluderede kodefiler
	tabsize=2, %
	extendedchars=true, % F.eks. æøå
	numbers=left, % Placering af linjenumre
	frame=l, % Ramme til venstre for listing
	framexleftmargin=-4pt,
	xleftmargin=5pt,
	breaklines=true,  % Linjeombrydning ved for lange linjer
	numberstyle=\scriptsize, % Style for numre
	basicstyle=\scriptsize, % Grundstyle for kode
	stepnumber=1, % Afstand mellem numre
	numbersep=0pt, % Afstand fra numre til margen ?
	keywordstyle=\bfseries\color{DTUred}, % Style for keywords (public, private mv.)
	commentstyle=\itshape\color{DTU11}, % Style for kommentarer (\\, /**/)
	stringstyle=\color{DTU9t} % Style for Strings ("")
}
